#include<stdio.h>
int main(){
    int i,j,n;
    printf("enter a value of n:");
    scanf("%d",&n);
    for(i=0;i<=n;i++){
        for(j=0;j<=n;j++){
            printf("%d",j);
        }
        printf("\n");
    }
    return 0;
